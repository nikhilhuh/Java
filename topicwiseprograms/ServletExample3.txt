Write a servlet program that takes username and password from a form using POST method and validates them.
If the username is “admin” and password is “12345”, display Login Successful, otherwise Invalid Credentials.

HTML Form (login.html) ->
<form action="LoginServlet" method="post">
  Username: <input type="text" name="username"><br><br>
  Password: <input type="password" name="password"><br><br>
  <input type="submit" value="Login">
</form>

Servlet Program (ServletExample3.java) ->
import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;

public class ServletExample3 extends HttpServlet {
    public void doPost(HttpServletRequest request, HttpServletResponse response)
        throws ServletException, IOException {

        response.setContentType("text/html");
        PrintWriter out = response.getWriter();

        String user = request.getParameter("username");
        String pass = request.getParameter("password");

        out.println("<html><body>");
        if (user.equals("admin") && pass.equals("12345")) {
            out.println("<h2 style='color:green;'>Login Successful!</h2>");
        } else {
            out.println("<h2 style='color:red;'>Invalid Credentials</h2>");
        }
        out.println("</body></html>");
    }
}
